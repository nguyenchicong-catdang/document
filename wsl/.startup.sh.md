# Cài đặt công cụ hỗ trợ expect
sudo apt install expect -y
# Viết nội dung cho file ~/.startup.sh
nano ~/.startup.sh
# vi du tham khao
```bash
#!/usr/bin/expect -f

# 1. Chạy lệnh ssh-add để thêm key
spawn ssh-add /home/cong/.ssh/id_ed25519

# 2. Chờ hệ thống hỏi mật khẩu (passphrase)
expect "Enter passphrase for /home/cong/.ssh/id_ed25519:"

# 3. Tự động gửi mật khẩu
send "MAT_KHAU_CUA_BAN\r"

# 4. Sau khi xong, trả lại quyền điều khiển cho người dùng
interact
```
# Cấp quyền và chạy thử
chmod +x ~/.startup.sh

## tham khao
```bash
#!/usr/bin/expect -f

# Tự động chạy cụm lệnh bash bên trong môi trường expect
spawn bash -c "cd ~ && ls; exec bash"

# Giữ tương tác để bạn có thể tiếp tục gõ lệnh
interact
```
