# learn
```bash
#!/bin/bash

# --- BƯỚC 1: ÉP NẠP MÔI TRƯỜNG ---
# Nạp bashrc để lấy PATH của npx/npm
. /home/cong/.bashrc

# Nạp NVM nếu bạn dùng nó
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"

# --- BƯỚC 2: XỬ LÝ SSH AGENT ---
# Khởi động agent và nạp key trong cùng 1 dòng để không mất connection
eval $(ssh-agent -s) && ssh-add /home/cong/.ssh/id_ed25519

# --- BƯỚC 3: THỰC THI LỆNH DỰ ÁN ---
echo "--- Đang khởi động dự án ---"
cd /home/cong/ws

# Mở VS Code
code ws.code-workspace

# Chạy server (Lệnh này sẽ giữ terminal để hiện log)
composer run dev

# --- BƯỚC 4: GIỮ TƯƠNG TÁC ---
exec bash
```
